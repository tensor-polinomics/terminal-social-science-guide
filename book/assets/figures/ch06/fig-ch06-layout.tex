% ======================================================================
% fig-ch06-layout.tex
% Chapter 6 (Organizing a project from the shell), the book's ONE Ch 6
% figure. Concept: a research project has one
% home per kind of file, and
% the layout encodes a one-directional flow. Fixed raw inputs (neutral)
% are READ by the scripts (blue, the spine, the code that does the work),
% which WRITE the regenerated products (green: cleaned data + output). A
% neutral band lists the tracked root files (the recipe + docs). The
% teaching claim: only raw inputs and code are precious; everything green
% is rebuildable (Chapter 12's `make`).
%
% Semantic palette (core three + neutral):
%   neutral = the given / fixed input and the tracked recipe files
%   blue    = scripts, the project spine (the code)
%   green   = regenerated products (data/clean, output)
%
% Build (self-contained; run from this directory; finds ../preamble.tex
% and, for icons, ../icons/cache/):
%   pdflatex fig-ch06-layout.tex
%   gs -o outlined.pdf -dNoOutputFonts -sDEVICE=pdfwrite fig-ch06-layout.pdf
%   pdftocairo -svg outlined.pdf fig-ch06-layout.svg
% ======================================================================
\documentclass[tikz,border=8pt]{standalone}
\input{../preamble.tex}
% lmodern gives a Type1 monospace (Latin Modern Mono) that
% outlines cleanly; the default TeX typewriter (cmtt) vanished
% from the HTML SVG. Build the SVG font-independently:
%   pdflatex fig-ch06-layout.tex
%   gs -o outlined.pdf -dNoOutputFonts -sDEVICE=pdfwrite \
%      fig-ch06-layout.pdf        # trace ALL text to paths
%   pdftocairo -svg outlined.pdf fig-ch06-layout.svg
% (dvisvgm --pdf produced a broken 1-node SVG here; do not use.)
\usepackage{lmodern}

\begin{document}
\begin{tikzpicture}[
  box/.style = {tbcard={#1}{center}, text width=34mm,
    minimum width=40mm, minimum height=17mm, inner sep=5pt,
    font=\sffamily\fontsize{8.5}{10.5}\selectfont,
    execute at begin node={\hyphenpenalty=10000\exhyphenpenalty=10000}},
  elabel/.style = {font=\sffamily\fontsize{7.5}{9}\selectfont,
    text=tbMute, fill=white, inner sep=1.5pt, align=center},
]

% --------------------------------------------------- the flow, left to right
\node[box={tbNeut}] (raw) at (0, 0)
  {\tbhead{database}{\ttfamily data/raw}\\[2pt]
   \tbdetail{fixed inputs, never edited by hand}};

\node[box={tbPrim}, tbstrong] (scripts) at (58mm, 0)
  {\tbhead{file-text}{\ttfamily scripts/}\\[2pt]
   \tbdetail{the numbered code, 00 to 04; the project spine}};

\node[box={tbData}] (clean) at (116mm, 12mm)
  {\tbhead{database}{\ttfamily data/clean}\\[2pt]
   \tbdetail{cleaned, derived data}};

\node[box={tbData}] (output) at (116mm, -12mm)
  {\tbhead{bar-chart-3}{\ttfamily output/}\\[2pt]
   \tbdetail{figures and tables}};

% --------------------------------------------------- flow edges
\draw[tbedge thick] (raw) -- (scripts)
  node[elabel, midway] {read};
\draw[tbedge thick] (scripts.east) -- (clean.west)
  node[elabel, pos=0.52] {write};
\draw[tbedge thick] (scripts.east) -- (output.west)
  node[elabel, pos=0.52] {write};

% --------------------------------------------------- regenerable bracket
\node[tbnote, text width=42mm] at (116mm, -30mm)
  {regenerated by one \texttt{make}\\(Chapter 12); safe to delete};

% --------------------------------------------------- tracked root files band
\node[neutL, text width=150mm, minimum height=12mm, inner sep=5pt,
  font=\sffamily\fontsize{8}{10}\selectfont] (root) at (58mm, -46mm)
  {\textbf{Project root, tracked in Git (the recipe and docs):}\\[1pt]
   {\ttfamily Makefile} $\cdot$ {\ttfamily report.qmd} $\cdot$
   {\ttfamily README.md} $\cdot$ {\ttfamily pyproject.toml}
   {\ttfamily +\,uv.lock} $\cdot$ {\ttfamily renv.lock} $\cdot$
   {\ttfamily .gitignore}};

\end{tikzpicture}
\end{document}
