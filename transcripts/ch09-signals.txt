The script under test (longjob.sh):

  #!/usr/bin/env bash
  cleanup() {
    echo "caught SIGTERM, cleaning up scratch files" >> job.log
    exit 0
  }
  trap cleanup TERM
  echo "job started (pid $$)" >> job.log
  while true; do sleep 0.2; done
---
